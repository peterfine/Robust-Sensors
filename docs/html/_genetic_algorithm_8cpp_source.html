00159 void GeneticAlgorithm::updateMutationVariance(double bestFitness) {
00160         if (myVariableMutationVariance) {
00161                 double proportionOfFitnessBetweenThresholds = (bestFitness
00162                                 - myVariableMutationVarianceMinFitnessThreshold)
00163                                 / (myVariableMutationVarianceMaxFitnessThreshold
00164                                                 - myVariableMutationVarianceMinFitnessThreshold);
00165 
00166                 // Clip to [0, 1]
00167                 if (proportionOfFitnessBetweenThresholds < 0.0)
00168                         proportionOfFitnessBetweenThresholds = 0.0;
00169                 if (proportionOfFitnessBetweenThresholds > 1.0)
00170                         proportionOfFitnessBetweenThresholds = 1.0;
00171 
00172                 // note 1 - proportionOfFitnessBetweenThresholds Is because max variance is low fitness and min is high fitness.
00173                 myMutationVariance = (myVariableMutationVarianceMax     - myVariableMutationVarianceMin) * 
00174                                                          (1     - proportionOfFitnessBetweenThresholds) 
00175                                                          + myVariableMutationVarianceMin;
