00006 // The Mersenne Twister is an algorithm for generating random numbers.  It
00007 // was designed with consideration of the flaws in various other generators.
00008 // The period, 2^19937-1, and the order of equidistribution, 623 dimensions,
00009 // are far greater.  The generator is also fast; it avoids multiplication and
00010 // division, and it benefits from caches and pipelines.  For more information
00011 // see the inventors' web page at http://www.math.keio.ac.jp/~matumoto/emt.html
00012

00170 inline double MTRand::randNorm( const double& mean, const double& variance )
00171 {
00172         // Return a real number from a normal (Gaussian) distribution with given
00173         // mean and variance by Box-Muller method
00174         double r = sqrt( -2.0 * log( 1.0-randDblExc()) ) * variance;
00175         double phi = 2.0 * 3.14159265358979323846264338328 * randExc();
00176         return mean + r * cos(phi);
00177 }

00179 inline MTRand::uint32 MTRand::randInt()
00180 {
00181         // Pull a 32-bit integer from the generator state
00182         // Every other access function simply transforms the numbers extracted here
00183         
00184         if( left == 0 ) reload();
00185         --left;
00186                 
00187         register uint32 s1;
00188         s1 = *pNext++;
00189         s1 ^= (s1 >> 11);
00190         s1 ^= (s1 <<  7) & 0x9d2c5680UL;
00191         s1 ^= (s1 << 15) & 0xefc60000UL;
00192         return ( s1 ^ (s1 >> 18) );
00193 }

00195 inline MTRand::uint32 MTRand::randInt( const uint32& n )
00196 {
00197         // Find which bits are used in n
00198         // Optimized by Magnus Jonsson (magnus@smartelectronix.com)
00199         uint32 used = n;
00200         used |= used >> 1;
00201         used |= used >> 2;
00202         used |= used >> 4;
00203         used |= used >> 8;
00204         used |= used >> 16;
00205         
00206         // Draw numbers until one is found in [0,n]
00207         uint32 i;
00208         do
00209                 i = randInt() & used;  // toss unused bits to shorten search
00210         while( i > n );
00211         return i;
00212 }

00223 inline void MTRand::seed( uint32 *const bigSeed, const uint32 seedLength )
00224 {
00225         // Seed the generator with an array of uint32's
00226         // There are 2^19937-1 possible initial states.  This function allows
00227         // all of those to be accessed by providing at least 19937 bits (with a
00228         // default seed length of N = 624 uint32's).  Any bits above the lower 32
00229         // in each element are discarded.
00230         // Just call seed() if you want to get array from /dev/urandom
00231         initialize(19650218UL);
00232         register int i = 1;
00233         register uint32 j = 0;
00234         register int k = ( N > seedLength ? N : seedLength );
00235         for( ; k; --k )
00236         {
00237                 state[i] =
00238                         state[i] ^ ( (state[i-1] ^ (state[i-1] >> 30)) * 1664525UL );
00239                 state[i] += ( bigSeed[j] & 0xffffffffUL ) + j;
00240                 state[i] &= 0xffffffffUL;
00241                 ++i;  ++j;
00242                 if( i >= N ) { state[0] = state[N-1];  i = 1; }
00243                 if( j >= seedLength ) j = 0;
00244         }
00245         for( k = N - 1; k; --k )
00246         {
00247                 state[i] =
00248                         state[i] ^ ( (state[i-1] ^ (state[i-1] >> 30)) * 1566083941UL );
00249                 state[i] -= i;
00250                 state[i] &= 0xffffffffUL;
00251                 ++i;
00252                 if( i >= N ) { state[0] = state[N-1];  i = 1; }
00253         }
00254         state[0] = 0x80000000UL;  // MSB is 1, assuring non-zero initial array
00255         reload();
00256 }

00259 inline void MTRand::seed()
00260 {
00261         // Seed the generator with an array from /dev/urandom if available
00262         // Otherwise use a hash of time() and clock() values
00263         
00264         // First try getting an array from /dev/urandom
00265         FILE* urandom = fopen( "/dev/urandom", "rb" );
00266         if( urandom )
00267         {
00268                 uint32 bigSeed[N];
00269                 register uint32 *s = bigSeed;
00270                 register int i = N;
00271                 register bool success = true;
00272                 while( success && i-- )
00273                         success = fread( s++, sizeof(uint32), 1, urandom );
00274                 fclose(urandom);
00275                 if( success ) { seed( bigSeed, N );  return; }
00276         }
00277         
00278         // Was not successful, so use time() and clock() instead
00279         seed( hash( time(NULL), clock() ) );
00280 }

00283 inline void MTRand::initialize( const uint32 seed )
00284 {
00285         // Initialize generator state with seed
00286         // See Knuth TAOCP Vol 2, 3rd Ed, p.106 for multiplier.
00287         // In previous versions, most significant bits (MSBs) of the seed affect
00288         // only MSBs of the state array.  Modified 9 Jan 2002 by Makoto Matsumoto.
00289         register uint32 *s = state;
00290         register uint32 *r = state;
00291         register int i = 1;
00292         *s++ = seed & 0xffffffffUL;
00293         for( ; i < N; ++i )
00294         {
00295                 *s++ = ( 1812433253UL * ( *r ^ (*r >> 30) ) + i ) & 0xffffffffUL;
00296                 r++;
00297         }
00298 }

00301 inline void MTRand::reload()
00302 {
00303         // Generate N new values in state
00304         // Made clearer and faster by Matthew Bellew (matthew.bellew@home.com)
00305         register uint32 *p = state;
00306         register int i;
00307         for( i = N - M; i--; ++p )
00308                 *p = twist( p[M], p[0], p[1] );
00309         for( i = M; --i; ++p )
00310                 *p = twist( p[M-N], p[0], p[1] );
00311         *p = twist( p[M-N], p[0], state[0] );
00312 
00313         left = N, pNext = state;
00314 }

00317 inline MTRand::uint32 MTRand::hash( time_t t, clock_t c )
00318 {
00319         // Get a uint32 from t and c
00320         // Better than uint32(x) in case x is floating point in [0,1]
00321         // Based on code by Lawrence Kirby (fred@genesis.demon.co.uk)
00322 
00323         static uint32 differ = 0;  // guarantee time-based seeds will change
00324 
00325         uint32 h1 = 0;
00326         unsigned char *p = (unsigned char *) &t;
00327         for( size_t i = 0; i < sizeof(t); ++i )
00328         {
00329                 h1 *= UCHAR_MAX + 2U;
00330                 h1 += p[i];
00331         }
00332         uint32 h2 = 0;
00333         p = (unsigned char *) &c;
00334         for( size_t j = 0; j < sizeof(c); ++j )
00335         {
00336                 h2 *= UCHAR_MAX + 2U;
00337                 h2 += p[j];
00338         }
00339         return ( h1 + differ++ ) ^ h2;
00340 }
