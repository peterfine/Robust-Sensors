00142 {

00144   // Skip delimiters at beginning.
00145   string::size_type lastPos = str.find_first_not_of(delimiters, 0);
00146   // Find first "non-delimiter".
00147   string::size_type pos     = str.find_first_of(delimiters, lastPos);
00148 
00149   while (string::npos != pos || string::npos != lastPos)
00150   {
00151     // Found a token, add it to the vector.
00152     tokens.push_back(str.substr(lastPos, pos - lastPos));
00153     // Skip delimiters.  Note the "not_of"
00154     lastPos = str.find_first_not_of(delimiters, pos);
00155     // Find next "non-delimiter"
00156     pos = str.find_first_of(delimiters, lastPos);
00157   }
