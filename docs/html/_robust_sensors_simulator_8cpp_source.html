00079 void RobustSensorsSimulator::update() {
00080 
00081         // Determine whether to move to the next light and sensor angle -----------------------------------------------------
00082         if(myTimeUntilNextLight <= 0.0) {
00083                 
00084                 // Check to see if the experiment has ended. If so, return.
00085                 if(myLightNo == (myNoLights - 1)) {
00086                         myExperimentHasEnded = true;
00087                         return;
00088                 }
00089                 
00090                 myLightNo++;
00091                 
00092                 // Now determine if the angle should be changed or not.
00093                 bool angleWasChanged = changeAngleIfRequired();
00094                 bool lightContributesToFitness = !angleWasChanged; // If the angle was changed, the light won't contribute.
00095                 
00096                 // Reset the light. Note that reset is also called in configure();
00097                 resetLight(lightContributesToFitness);
00098         }
00099         myTimeUntilNextLight -= myTimestep;
00100 
00101         // Calculate Sensor Inputs -------------------------------------------------------------------------
00102         // Calculate the sensor locations.
00103         double agentDistSq   = pow(myLightX - myAgentX, 2) + pow(myLightY - myAgentY, 2);
00104         double agentDist = sqrt(agentDistSq);
00105 
00106         mySensor1X = myAgentX + myRadius * cos(myAgentHeading + mySensor1Angle);
00107         mySensor1Y = myAgentY + myRadius * sin(myAgentHeading + mySensor1Angle);
00108         double sensor1DistSq = pow(myLightX - mySensor1X, 2) + pow(myLightY - mySensor1Y, 2); 
00109         double sensor1Dist = sqrt(sensor1DistSq);
00110         
00111                         
00112         // Calculate whether the light impinges on the sensor 1.
00113         mySensor1Input = 0.0;
00114         if((pow(myRadius, 2) + sensor1DistSq) / agentDistSq <= 1) {
00115                 // Calculate the linear sensor input.
00116                 if(sensor1Dist < myLightDistanceMax * 1.5) {
00117                         mySensor1Input = (1 - sensor1Dist / (myLightDistanceMax * 1.5));
00118                 }
00119         }
00120         
00121         myNetwork.setInput(0, mySensor1Input); 
00122         
00123         // Update the network and calculate the motor outputs ----------------------------------------------------------------
00124         
00125         myNetwork.updateNetwork(myTimestep);
00126 
00127         double moveV = (myNetwork.getOutput(0) + myNetwork.getOutput(1)) / 2;
00128         double turnV = (myNetwork.getOutput(0) - myNetwork.getOutput(1)) / (2 * myRadius);
00129         
00130         myAgentX += myTimestep * moveV * cos(myAgentHeading);
00131         myAgentY += myTimestep * moveV * sin(myAgentHeading);
00132         myAgentHeading += myTimestep * turnV;
00133         
00134         // Only add this to the fitness value if...
00135         if(myLightContributesToFitness && // This light is not part of the initial phase (or after a change of angle). 
00136                  (myTimeUntilNextLight <= myFitnessDuration)) { // The time is within the fitness contributing phase of the current light
00137                 double currentFitness = 0.0;
00138                 if(agentDist < myLightDistanceMax) {
00139                         currentFitness = 1.0 - (agentDist / myLightDistanceMax);
00140                 }
00141                 
00142                 myUnNormalizedFitness += currentFitness * myTimestep;
00143                 myTotalFitnessTime += myTimestep;
00144         }
00145         
00146         myTime += myTimestep;
00147         
00148 }
